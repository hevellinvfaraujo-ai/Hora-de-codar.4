programa
{
	
	funcao inicio()
	{
		// Variáveis
		cadeia nome, extrato_historico
		real saldo = 1500.00, valor
		inteiro opcao, senha, conta_destino
		
		// Inicialização
		extrato_historico = "Extrato:\n" +
						  "- Depósito de R$ 500.00\n" +
						  "- Compra em Farmácia: -R$ 45.50\n" +
						  "- Depósito de R$ 100.00\n"

		// 1. Mensagem de Boas-vindas
		escreva("Por favor, digite seu nome: ")
		leia(nome)
		escreva("Olá ", nome, ", é um prazer ter você por aqui!\n\n")

		// Loop principal do programa
		faca
		{
			// 10. Validação de Senha para acesso às operações
			escreva("Digite a sua senha: ")
			leia(senha)

			// Caso a senha esteja incorreta, pede novamente
			se (senha != 3589) {
				escreva("Senha incorreta. Tente novamente.\n\n")
			// Volta para o início do loop
			}

			// Exibe o menu principal
			escreva("\n")
			escreva("Bem-vindo(a) ao menu principal, ", nome, "!\n")
			escreva("1. Saldo\n")
			escreva("2. Extrato\n")
			escreva("3. Saque\n")
			escreva("4. Depósito\n")
			escreva("5. Transferência\n")
			escreva("6. Sair\n")
			escreva("Escolha uma opção: ")
			leia(opcao)

			// 2. Usando escolha/caso (switch/case) para as opções
			escolha (opcao)
			{
				caso 1:
					escreva("Seu saldo atual é de R$ ", saldo, "\n\n")
					pare
				caso 2:
					escreva(extrato_historico, "\n")
					escreva("Saldo atual: R$ ", saldo, "\n\n")
					pare
				caso 3:
					escreva("Digite o valor para saque: ")
					leia(valor)

					// 3. e 4. Validações de saque (valor e saldo)
					se (valor <= 0) {
						escreva("Operação não autorizada. O valor deve ser maior que zero.\n\n")
					} senao se (valor > saldo) {
						escreva("Operação não autorizada. Saldo insuficiente.\n\n")
					} senao {
						saldo = saldo - valor
						escreva("Saque de R$ ", valor, " realizado com sucesso.\n")
						escreva("Novo saldo: R$ ", saldo, "\n\n")
					}
					pare
				caso 4:
					escreva("Digite o valor para depósito: ")
					leia(valor)

					// 11. Validação de depósito
					se (valor <= 0) {
						escreva("Operação não autorizada. O valor deve ser maior que zero.\n\n")
					} senao {
						saldo = saldo + valor
						escreva("Depósito de R$ ", valor, " realizado com sucesso.\n")
						escreva("Novo saldo: R$ ", saldo, "\n\n")
						extrato_historico = extrato_historico + "- Depósito de R$ " + valor + "\n"
					}
					pare
				caso 5:
					escreva("Digite o número da conta de destino: ")
					leia(conta_destino)
					escreva("Digite o valor para transferência: ")
					leia(valor)

					// 6. e 7. Validações de transferência
					se (valor <= 0) {
						escreva("Operação não autorizada. O valor deve ser maior que zero.\n\n")
					} senao se (valor > saldo) {
						escreva("Operação não autorizada. Saldo insuficiente para transferência.\n\n")
					} senao {
						saldo = saldo - valor
						escreva("Transferência de R$ ", valor, " para a conta ", conta_destino, " realizada com sucesso.\n")
						escreva("Novo saldo: R$ ", saldo, "\n\n")
						extrato_historico = extrato_historico + "- Transferência para conta " + conta_destino + ": -R$ " + valor + "\n"
					}
					pare
				caso 6:
					// 12. Mensagem de despedida
					escreva("\n", nome, ", foi um prazer ter você por aqui! Até mais.\n")
					pare
				
				caso contrario:
					// 9. Atualização da função "erro"
					escreva("Opção inválida. Por favor, escolha uma das opções do menu.\n\n")
					
			}
		} enquanto (opcao != 6)
	}
}
